 77 
 78             var worldBody=          world.CreateBody(bodyDef);
 79             var worldBodyFixture=   worldBody.CreateFixture(fixDef);
 80 
 81             if ( bodyData.isSensor ) {
 82                 worldBodyFixture.SetSensor(true);
 83             }
 84 
 85             this.worldBody=         worldBody;
 86             this.worldBodyFixture=  worldBodyFixture;
 87             this.fixtureDef=        fixDef;
 88             this.bodyDef=           bodyDef;
 89             this.bodyData=          bodyData;
 90 
 91 
 92             this.setFillStyle(this.worldBodyFixture.IsSensor() ? 'red' : 'blue').
 93                     setBackgroundImage(this.image).
 94                     setSize(2*this.radius,2*this.radius).

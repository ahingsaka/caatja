  7 CAAT.Module({
  8     defines:"CAAT.Module.Audio.AudioManager",
  9     depends:[
 10         "CAAT.Module.Runtime.BrowserInfo"
 11     ],
 12     extendsWith:function () {
 13         return {
 14             __init:function () {
 15                 this.browserInfo = CAAT.Module.Runtime.BrowserInfo;
 16                 return this;
 17             },
 18 
 19             browserInfo:null,
 20             musicEnabled:true,
 21             fxEnabled:true,
 22             audioCache:null, // audio elements.
 23             channels:null, // available playing channels.
 24             workingChannels:null, // currently playing channels.
 25             loopingChannels:[],
 26             audioTypes:{               // supported audio formats. Don't remember where i took them from :S
 27                 'mp3':'audio/mpeg;',
 28                 'ogg':'audio/ogg; codecs="vorbis"',
 29                 'wav':'audio/wav; codecs="1"',
 30                 'mp4':'audio/mp4; codecs="mp4a.40.2"'
 31             },
 32 
 33             /**
 34              * Initializes the sound subsystem by creating a fixed number of Audio channels.
 35              * Every channel registers a handler for sound playing finalization. If a callback is set, the
 36              * callback function will be called with the associated sound id in the cache.
 37              *
 38              * @param numChannels {number} number of channels to pre-create. 8 by default.
 39              *
 40              * @return this.
 41              */
 42             initialize:function (numChannels) {
 43 
 44                 this.audioCache = [];
 45                 this.channels = [];
 46                 this.workingChannels = [];
 47 
 48                 for (var i = 0; i < numChannels; i++) {
 49                     var channel = document.createElement('audio');
 50 
 51                     if (null !== channel) {
 52                         channel.finished = -1;
 53                         this.channels.push(channel);
 54                         var me = this;
 55                         channel.addEventListener(
 56                             'ended',
 57                             // on sound end, set channel to available channels list.
 58                             function (audioEvent) {
 59                                 var target = audioEvent.target;
 60                                 var i;
 61 
 62                                 // remove from workingChannels
 63                                 for (i = 0; i < me.workingChannels.length; i++) {
 64                                     if (me.workingChannels[i] === target) {
 65                                         me.workingChannels.splice(i, 1);
 66                                         break;
 67                                     }
 68                                 }
 69 
 70                                 if (target.caat_callback) {
 71                                     target.caat_callback(target.caat_id);
 72                                 }
 73 
 74                                 // set back to channels.
 75                                 me.channels.push(target);
 76                             },
 77                             false
 78                         );
 79                     }
 80                 }
 81 
 82                 return this;
 83             },
 84             /**
 85              * Tries to add an audio tag to the available list of valid audios. The audio is described by a url.
 86              * @param id {object} an object to associate the audio element (if suitable to be played).
 87              * @param url {string} a string describing an url.
 88              * @param endplaying_callback {function} callback to be called upon sound end.
 89              *
 90              * @return {boolean} a boolean indicating whether the browser can play this resource.
 91              *
 92              * @private
 93              */
 94             addAudioFromURL:function (id, url, endplaying_callback) {
 95                 var extension = null;
 96                 var audio = document.createElement('audio');
 97 
 98                 if (null !== audio) {
 99 
100                     if (!audio.canPlayType) {
101                         return false;
102                     }
103 
104                     extension = url.substr(url.lastIndexOf('.') + 1);
105                     var canplay = audio.canPlayType(this.audioTypes[extension]);
106 
107                     if (canplay !== "" && canplay !== "no") {
108                         audio.src = url;
109                         audio.preload = "auto";
110                         audio.load();
111                         if (endplaying_callback) {
112                             audio.caat_callback = endplaying_callback;
113                             audio.caat_id = id;
114                         }
115                         this.audioCache.push({ id:id, audio:audio });
116 
117                         return true;
118                     }
119                 }
120 
121                 return false;
122             },
123             /**
124              * Tries to add an audio tag to the available list of valid audios. The audio element comes from
125              * an HTMLAudioElement.
126              * @param id {object} an object to associate the audio element (if suitable to be played).
127              * @param audio {HTMLAudioElement} a DOM audio node.
128              * @param endplaying_callback {function} callback to be called upon sound end.
129              *
130              * @return {boolean} a boolean indicating whether the browser can play this resource.
131              *
132              * @private
133              */
134             addAudioFromDomNode:function (id, audio, endplaying_callback) {
135 
136                 var extension = audio.src.substr(audio.src.lastIndexOf('.') + 1);
137                 if (audio.canPlayType(this.audioTypes[extension])) {
138                     if (endplaying_callback) {
139                         audio.caat_callback = endplaying_callback;
140                         audio.caat_id = id;
141                     }
142                     this.audioCache.push({ id:id, audio:audio });
143 
144                     return true;
145                 }
146 
147                 return false;
148             },
149             /**
150              * Adds an elements to the audio cache.
151              * @param id {object} an object to associate the audio element (if suitable to be played).
152              * @param element {URL|HTMLElement} an url or html audio tag.
153              * @param endplaying_callback {function} callback to be called upon sound end.
154              *
155              * @return {boolean} a boolean indicating whether the browser can play this resource.
156              *
157              * @private
158              */
159             addAudioElement:function (id, element, endplaying_callback) {
160                 if (typeof element === "string") {
161                     return this.addAudioFromURL(id, element, endplaying_callback);
162                 } else {
163                     try {
164                         if (element instanceof HTMLAudioElement) {
165                             return this.addAudioFromDomNode(id, element, endplaying_callback);
166                         }
167                     }
168                     catch (e) {
169                     }
170                 }
171 
172                 return false;
173             },
174             /**
175              * creates an Audio object and adds it to the audio cache.
176              * This function expects audio data described by two elements, an id and an object which will
177              * describe an audio element to be associated with the id. The object will be of the form
178              * array, dom node or a url string.
179              *
180              * <p>
181              * The audio element can be one of the two forms:
182              *
183              * <ol>
184              *  <li>Either an HTMLAudioElement/Audio object or a string url.
185              *  <li>An array of elements of the previous form.
186              * </ol>
187              *
188              * <p>
189              * When the audio attribute is an array, this function will iterate throught the array elements
190              * until a suitable audio element to be played is found. When this is the case, the other array
191              * elements won't be taken into account. The valid form of using this addAudio method will be:
192              *
193              * <p>
194              * 1.<br>
195              * addAudio( id, url } ). In this case, if the resource pointed by url is
196              * not suitable to be played (i.e. a call to the Audio element's canPlayType method return 'no')
197              * no resource will be added under such id, so no sound will be played when invoking the play(id)
198              * method.
199              * <p>
200              * 2.<br>
201              * addAudio( id, dom_audio_tag ). In this case, the same logic than previous case is applied, but
202              * this time, the parameter url is expected to be an audio tag present in the html file.
203              * <p>
204              * 3.<br>
205              * addAudio( id, [array_of_url_or_domaudiotag] ). In this case, the function tries to locate a valid
206              * resource to be played in any of the elements contained in the array. The array element's can
207              * be any type of case 1 and 2. As soon as a valid resource is found, it will be associated to the
208              * id in the valid audio resources to be played list.
209              *
210              * @return this
211              */
212             addAudio:function (id, array_of_url_or_domnodes, endplaying_callback) {
213 
214                 if (array_of_url_or_domnodes instanceof Array) {
215                     /*
216                      iterate throught array elements until we can safely add an audio element.
217                      */
218                     for (var i = 0; i < array_of_url_or_domnodes.length; i++) {
219                         if (this.addAudioElement(id, array_of_url_or_domnodes[i], endplaying_callback)) {
220                             break;
221                         }
222                     }
223                 } else {
224                     this.addAudioElement(id, array_of_url_or_domnodes, endplaying_callback);
225                 }
226 
227                 return this;
228             },
229             /**
230              * Returns an audio object.
231              * @param aId {object} the id associated to the target Audio object.
232              * @return {object} the HTMLAudioElement addociated to the given id.
233              */
234             getAudio:function (aId) {
235                 for (var i = 0; i < this.audioCache.length; i++) {
236                     if (this.audioCache[i].id === aId) {
237                         return this.audioCache[i].audio;
238                     }
239                 }
240 
241                 return null;
242             },
243 
244             /**
245              * Set an audio object volume.
246              * @param id {object} an audio Id
247              * @param volume {number} volume to set. The volume value is not checked.
248              *
249              * @return this
250              */
251             setVolume:function (id, volume) {
252                 var audio = this.getAudio(id);
253                 if (null != audio) {
254                     audio.volume = volume;
255                 }
256 
257                 return this;
258             },
259 
260             /**
261              * Plays an audio file from the cache if any sound channel is available.
262              * The playing sound will occupy a sound channel and when ends playing will leave
263              * the channel free for any other sound to be played in.
264              * @param id {object} an object identifying a sound in the sound cache.
265              * @return this.
266              */
267             play:function (id) {
268                 if (!this.fxEnabled) {
269                     return this;
270                 }
271 
272                 var audio = this.getAudio(id);
273                 // existe el audio, y ademas hay un canal de audio disponible.
274                 if (null !== audio && this.channels.length > 0) {
275                     var channel = this.channels.shift();
276                     channel.src = audio.src;
277                     channel.load();
278                     channel.volume = audio.volume;
279                     channel.play();
280                     this.workingChannels.push(channel);
281                 }
282 
283                 return this;
284             },
285             /**
286              * This method creates a new AudioChannel to loop the sound with.
287              * It returns an Audio object so that the developer can cancel the sound loop at will.
288              * The user must call <code>pause()</code> method to stop playing a loop.
289              * <p>
290              * Firefox does not honor the loop property, so looping is performed by attending end playing
291              * event on audio elements.
292              *
293              * @return {HTMLElement} an Audio instance if a valid sound id is supplied. Null otherwise
294              */
295             loop:function (id) {
296 
297                 if (!this.musicEnabled) {
298                     return this;
299                 }
300 
301                 var audio_in_cache = this.getAudio(id);
302                 // existe el audio, y ademas hay un canal de audio disponible.
303                 if (null !== audio_in_cache) {
304                     var audio = document.createElement('audio');
305                     if (null !== audio) {
306                         audio.src = audio_in_cache.src;
307                         audio.preload = "auto";
308 
309                         if (this.browserInfo.browser === 'Firefox') {
310                             audio.addEventListener(
311                                 'ended',
312                                 // on sound end, set channel to available channels list.
313                                 function (audioEvent) {
314                                     var target = audioEvent.target;
315                                     target.currentTime = 0;
316                                 },
317                                 false
318                             );
319                         } else {
320                             audio.loop = true;
321                         }
322                         audio.load();
323                         audio.play();
324                         this.loopingChannels.push(audio);
325                         return audio;
326                     }
327                 }
328 
329                 return null;
330             },
331             /**
332              * Cancel all playing audio channels
333              * Get back the playing channels to available channel list.
334              *
335              * @return this
336              */
337             endSound:function () {
338                 var i;
339                 for (i = 0; i < this.workingChannels.length; i++) {
340                     this.workingChannels[i].pause();
341                     this.channels.push(this.workingChannels[i]);
342                 }
343 
344                 for (i = 0; i < this.loopingChannels.length; i++) {
345                     this.loopingChannels[i].pause();
346                 }
347 
348                 return this;
349             },
350             setSoundEffectsEnabled:function (enable) {
351                 this.fxEnabled = enable;
352                 return this;
353             },
354             isSoundEffectsEnabled:function () {
355                 return this.fxEnabled;
356             },
357             setMusicEnabled:function (enable) {
358                 this.musicEnabled = enable;
359                 for (var i = 0; i < this.loopingChannels.length; i++) {
360                     if (enable) {
361                         this.loopingChannels[i].play();
362                     } else {
363                         this.loopingChannels[i].pause();
364                     }
365                 }
366                 return this;
367             },
368             isMusicEnabled:function () {
369                 return this.musicEnabled;
370             }
371         }
372     }
373 });
